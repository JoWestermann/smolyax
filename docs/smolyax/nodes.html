 11class Leja1D(Generator1D):
 12    """
 13    Generator for one-dimensional Leja nodes.
 14
 15    Produces nested Leja nodes (default on [-1, 1]), with optional scaling to arbitrary bounded domains.
 16    """
 17
 18    __nodes = np.array([0, 1, -1, 1 / np.sqrt(2), -1 / np.sqrt(2)])
 19
 20    def __init__(self, domain: Union[jax.Array, np.ndarray, Sequence[float]] = None) -> None:
 21        """
 22        Initialize the one-dimensional Leja node generator.
 23
 24        Parameters
 25        ----------
 26        domain : Union[jax.Array, np.ndarray, Sequence[float]], optional
 27            Endpoints of the domain. Defaults to `[-1, 1]` if not specified.
 28        """
 29        super().__init__(is_nested=True)
 30        self.__domain = domain
 31        self.__reference_domain = None
 32        if domain is not None:
 33            self.__domain = np.asarray(domain)
 34            self.__reference_domain = (-1, 1)
 35        self.__cached_call = self.__make_cached_call()
 36        self.__cached_get_quadrature_weights = self.__make_cached_get_quadrature_weights()
 37
 38    @property
 39    def domain(self) -> Union[jax.Array, np.ndarray, Sequence[float]]:
 40        """
 41        Domain interval of the node sequence.
 42
 43        Returns
 44        -------
 45        array-like of shape (2,)
 46        """
 47        return self.__domain
 48
 49    @classmethod
 50    def __ensure_nodes(cls, n: int):
 51        k = cls.__nodes.shape[0]
 52        if n >= k:
 53            cls.__nodes = np.append(cls.__nodes, np.empty((n + 1 - k,)))
 54            for j in range(k, n + 1):
 55                if j % 2 == 0:
 56                    cls.__nodes[j] = -cls.__nodes[j - 1]
 57                else:
 58                    cls.__nodes[j] = np.sqrt((cls.__nodes[int((j + 1) / 2)] + 1) / 2)
 59
 60    def __make_cached_call(self):
 61        @lru_cache(maxsize=None)
 62        def cached(n):
 63            self.__ensure_nodes(n)
 64            return self.scale(self.__nodes[: n + 1])
 65
 66        return cached
 67
 68    def __call__(self, n: int) -> Union[jax.Array, np.ndarray]:
 69        """
 70        Generate Leja nodes mapped to `self.domain`.
 71
 72        Parameters
 73        ----------
 74        n : int
 75            Maximal node index (counting starts at `0`).
 76
 77        Returns
 78        -------
 79        jax.Array or np.ndarray
 80            Leja nodes mapped to the `self.domain`.
 81        """
 82        return self.__cached_call(n)
 83
 84    def __make_cached_get_quadrature_weights(self):
 85        @lru_cache(maxsize=None)
 86        def cached(n):
 87            self.__ensure_nodes(n)
 88            quadrature_points = self.__nodes[: n + 1]
 89            vm_matrix = np.vstack([quadrature_points**i for i in range(n + 1)])
 90            rhs = np.array([(1 + (-1) ** i) / (2.0 * (i + 1)) for i in range(n + 1)])
 91            quadrature_weights = np.linalg.solve(vm_matrix, rhs)
 92            return quadrature_weights
 93
 94        return cached

 96    def scale(
 97        self,
 98        x: Union[jax.Array, np.ndarray],
 99        d1: Union[jax.Array, np.ndarray, Sequence[float]] = None,
100        d2: Union[jax.Array, np.ndarray, Sequence[float]] = None,
101    ) -> Union[jax.Array, np.ndarray]:
102        """
103        Map nodes or points from interval `d1` to interval `d2` by affine transformation.
104
105        Mainly intended for internal use and testing.
106
107        Parameters
108        ----------
109        x : array-like
110            Points to transform.
111        d1 : array-like of shape (2,), optional
112            Source interval (default: reference domain `[-1, 1]`).
113        d2 : array-like of shape (2,), optional
114            Target interval (default: generator's domain).
115
116        Returns
117        -------
118        jax.Array or np.ndarray
119            Nodes or points in the target domain.
120        """
121        if d1 is None:
122            d1 = self.__reference_domain
123        if d2 is None:
124            d2 = self.__domain
125
126        assert (d1 is None) == (d2 is None)
127        if d1 is None:  # no scaling if no custom domains are give
128            return x
129
130        # ensure d1, d2 have shape (2, )
131        d1, d2 = np.squeeze(d1), np.squeeze(d2)
132        assert d1.shape == d2.shape == (2,), f"shapes {d1.shape} and {d2.shape} do not match (2, )"
133        assert d1[0] < d1[1]
134        assert d2[0] < d2[1]
135
136        # ensure x has shape (n, )
137        x_shape = x.shape
138        x = np.squeeze(x)
139        assert x.ndim <= 1
140
141        # ensure x in d1
142        valid_lower = (x >= d1[0]) | np.isclose(x, d1[0])
143        valid_upper = (x <= d1[1]) | np.isclose(x, d1[1])
144        assert np.all(valid_lower), f"Assertion failed: Some values are below lower bounds\n{x[~valid_lower]}"
145        assert np.all(valid_upper), f"Assertion failed: Some values are above upper bounds\n{x[~valid_upper]}"
146
147        # scale
148        x = (x - d1[0]) / (d1[1] - d1[0])
149        x = x * (d2[1] - d2[0]) + d2[0]
150
151        # Return in original shape
152        return x.reshape(x_shape)
153
